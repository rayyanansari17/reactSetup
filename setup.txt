git clone your project repo 
cd into repo name 
mkdir client && cd client 
pnpm create vite@latest .

choose:
framework: React
varient: js

delete:
rm -rf eslint.config.js src/assets/react.svg src/App.css src/App.jsx src/index.css src/main.jsx

recreate:
touch src/App.css src/App.jsx src/index.css src/main.jsx

pnpm install && pnpm up --latest
pnpm add tailwindcss @tailwindcss/vite

in src/index.css write:
@import 'tailwindcss'

inside vite.config.jsx write 
import { defineConfig } from 'vite'
import react from 'vitejs/plugin-react'
import tailwindcss from '@tailwindcss/vite'

export default defineConfig({
    plugin: [react(), tailwindcss()]
})

inside src/main.jsx write 

import { creareRoot } from 'react-dom/client';

createRoot (document.getElementById('root')).render(<App />);
